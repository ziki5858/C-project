68 void data_def(struct Node *current_pattern, int current_pattern_num, int *DC, int *error_flag, int *IC) {
69  int sign = 1, i, data_element;
70  char *buffer, *temp_data;
71  /* check if there is a label */
72  if (current_pattern->data->label[0]) {
73  handle_lable_define(current_pattern->data, "data", *DC,
74  current_pattern_num, error_flag, *IC, *DC, current_pattern->data->choice.dir.size);
75  }
76  /* insert the data to the data table */
77  allocate_memory_for_data(current_pattern->data, current_pattern_num, error_flag);
78  for (i = 0; i < current_pattern->data->choice.dir.size; i++) {
79  buffer = current_pattern->data->choice.dir.data[i];
80 
81  if (isalpha(buffer[0]) &&
82  exist_in_trie(constant, buffer)) { /* if the data is a constant */
83  data_element = ((Constant)exist_in_trie(constant, buffer))->value;
84  } else { /* if the data is a number */
85  if (buffer[0] == '-' || buffer[0] == '+') { /* ignore the sign */
86  sign = determaine_sign(buffer); /* determaine the sign */
87  buffer++;
88  }
89  if (!containsOnlyDigits(&buffer[0])) { /* check if the data is a number */
90  print_error_msg("data element is not a number", current_pattern_num, error_flag);
91  return;
92  } else
93  data_element = atoi(buffer) * sign;
94  }
95  if (data_element > MAX_DATA_VALUE || data_element < MIN_DATA_VALUE) {
96  print_error_msg("data element is out of range", current_pattern_num, error_flag);
97  return;
98  }
99  /* insert the data to the data table */
100  temp_data = encoded_data(data_element);
101  strcpy(data->lines[data->num_of_lines + i]->word, temp_data); /* -> */
102  free(temp_data);
103  }
104  data->num_of_lines += current_pattern->data->choice.dir.size;
105  (*DC) += current_pattern->data->choice.dir.size;
106 }
